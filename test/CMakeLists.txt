cmake_minimum_required(VERSION 3.2)

include_directories(${GTEST_INCLUDE_DIRS})

    fhe

    ntl

    boost_system
    gmp)

set(CPP_ITESTS
    test_net_io
    test_multithread_server
	test_mult_in_slots

    target_link_libraries(${CPP_ITEST} ${RUNTIME_LIBS} ${GTEST_BOTH_LIBRARIES})
endforeach(CPP_ITEST)
